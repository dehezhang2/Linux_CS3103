"Quick Run By F5
    map <F5> :call CompileRunGcc()<CR>

    func! CompileRunGcc()
        exec "w" 
        if &filetype == 'c' 
            exec '!g++ % -o %<'
            exec '!time ./%<'
        elseif &filetype == 'cpp'
            exec '!g++ % -o %<'
            exec '!time ./%<'
        elseif &filetype == 'python'
            exec '!time python %'
        elseif &filetype == 'sh'
            :!time bash %
        endif                                                                              
    endfunc
